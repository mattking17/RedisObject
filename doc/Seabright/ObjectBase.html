def initialize(ident={}, prnt = nil)
        if prnt && prnt.class != String
                @parent = prnt
                @parent_id = prnt.hkey
        else
                @parent_id = prnt
        end
        if ident && ident.class == String
                load(ident)
        elsif ident && ident.class == Hash
                ident[id_sym] ||= generate_id
                if load(ident[id_sym])
                        @data = ident
                end
        end
end

def dump(prnt=nil)
        require "utf8_utils"
        out = ["puts \"Creating: #{id}\""]
        s_id = (prnt ? "#{prnt.id} #{id}" : id).gsub(%r\W/,"_")
        out << "a#{s_id} = #{self.class.cname}.new(#{actual.to_s.tidy_bytes},#{prnt.id})"
        out << "a#{prnt.id.gsub(/\W/,"_")} << a#{s_id}" if prnt
        out << "a#{s_id}.save"
        out.join("\n")
end

def generate_id
        v = new_id
        while self.class.exists?(v) do
                puts "[RedisObject] Collision at id: #{v}" if DEBUG
                v = new_id
        end
        puts "[RedisObject] Reserving key: #{v}" if DEBUG
        reserve(v)
        v
end

def save
        set(:class, self.class.name)
        set(id_sym,id)
        set(:key, key)
        if @data
                saving = @data
                @data = false
                saving.each do |k,v|
                        set(k,v)
                end
        end
        store.sadd(self.class.cname.pluralize, key)
        store.del(reserve_key)
end
