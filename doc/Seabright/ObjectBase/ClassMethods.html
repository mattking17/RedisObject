def deep_const_get(const)
        if Symbol === const
                const = const.to_s
        else
                const = const.to_str.dup
        end
        if const.sub!(%r^::/, '')
                base = Object
        else
                base = self
        end
        const.split(%r::/).inject(base) { |mod, name| mod.const_get(name) }
end

def find(ident,prnt=nil)
        if ident.class == String
                return store.exists(self.hkey(ident,prnt)) ? self.new(ident,prnt) : nil
        elsif ident.class == Hash
                each do |obj|
                        good = true
                        ident.each do |k,v|
                                good = false if !obj[k.to_sym] || obj[k.to_sym]!=v
                        end
                        return obj if good
                end
        end
        nil
end

def find_by_key(k)
        if cls = store.hget(k,:class) 
                o_id = store.hget(k,id_sym(cls))
                prnt = store.hget(k,:parent)
                if store.exists(k)
                        return deep_const_get(cls.to_sym).new(o_id,prnt)
                end
        end
        nil
end
