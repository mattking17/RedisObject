def dump(prnt=nil)
        require "utf8_utils"
        out = ["puts \"Creating: #{id}\""]
        s_id = (prnt ? "#{prnt.id} #{id}" : id).gsub(%r\W/,"_")
        out << "a#{s_id} = #{self.class.cname}.new(#{actual.to_s.tidy_bytes},#{prnt.id})"
        @collections.each do |col|
                col.each do |sobj|
                        out << sobj.dump(self)
                end
        end
        out << "a#{prnt.id.gsub(/\W/,"_")} << a#{s_id}" if prnt
        out << "a#{s_id}.save"
        out.join("\n")
end
