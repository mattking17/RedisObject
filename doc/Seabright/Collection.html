def indexed(index,num=5,reverse=false)
        out = []
        store.send(reverse ? :zrevrange : :zrange,index_key(index), 0, num).each do |member|
                if a = class_const.find_by_key(member)
                        out << a
                else
                        store.zrem(index_key(index),member)
                end
        end
        out
end
