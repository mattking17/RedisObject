def self.included(base)
        @@sort_indices ||= []
        @@sort_indices.push(:created_at)
        @@sort_indices.push(:updated_at)
        @@field_formats ||= {}
        @@field_formats[:created_at] = :format_date
        @@field_formats[:updated_at] = :format_date
        @@time_irrelevant = false
        base.extend(ClassMethods)
end

def update_timestamps
        return if @@time_irrelevant
        set(:created_at, Time.now) if !get(:created_at)
        set(:updated_at, Time.now)
end
